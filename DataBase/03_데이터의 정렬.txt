/*
  데이터의 정렬
    기본형태
  SELECT *|{DISTINCT} column|표현식 [별칭],...}
  FROM table
  [WHERE 조건(s)]
  [ORDER BY {column, 표현식} [ASC|DESC]];
*/
/*
1. 오름차순 정렬
문자는 영문자 순 한글은 가나다 순 정렬 
ASC절은 생략가능
*/
-- 사원정보를 입사일 순으로 정렬
SELECT emp_name, position, dept_id, hiredate
FROM y_emp
ORDER BY hiredate;

/*
2. 내림차순 정렬
역순으로 정리할땐 ORDER BY 절에서 열이름 다음에 DESC 키워드 지정
*/
-- 최근 입사한 사원 순 결과 정렬
SELECT emp_name, position, dept_id, hiredate
FROM y_emp
ORDER BY hiredate DESC;

/*
3. 여러 열 기준의 정렬
여러 열을 기준으로 쿼리 결과를 정렬 가능. > 테이블의 열 수 만큼 가능
ORDER BY 절에 쉼표로 열 이름을 구분하면서 지정
ASC, DESC는 열 이름마다 별도로 지정
SELECT 절에 포함되지 않은 열을 기준으로도 정렬 가능
*/
-- 사원번호, 이름, 부서번호, 급여 출력
-- 부서번호 기준으로 오름차순 정렬 (오름차순)
-- 동일 부서내 급여 순으로 정렬 (내림차순)
SELECT emp_id, emp_name, dept_id, salary
FROM y_emp
ORDER BY dept_id, salary DESC;

/*
4. 별칭 또는 위치 값 기준의 정렬
ORDER BY 절에 지정하는 정렬 조건으로 표현식, 별칭 열 위치값 등을 지정 할수 있다.
*/
-- 사원번호, 이름, 연봉(SALARY *12) 조회
-- 연봉을 내림차순으로 정렬
SELECT emp_id, emp_name, salary * 12 annsal
FROM y_emp
ORDER BY annsal DESC;

-- 위 예제에서 열 또는 표현식의 위치값으로 대체 가능
-- 3번 위치에 있는 연봉이 내림차순으로 정렬된다.
SELECT emp_id, emp_name, salary * 12 annsal
FROM y_emp
ORDER BY 3 DESC;
